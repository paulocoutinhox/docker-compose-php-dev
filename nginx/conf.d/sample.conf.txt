server {
    listen  80;
    listen  443 ssl;

    server_name  localhost;

    charset UTF-8;

	set $bootstrap "index.php";

	root /usr/share/nginx/html;

	location / {		        
        index index.html index.htm $bootstrap;
		try_files $uri $uri/ /$bootstrap?$args;
    }

    # redirect server not found error to the static page /404.html
    #
	error_page  404 /404.html;

    # redirect server error pages to the static page /50x.html
    #
	error_page 500 502 503 504 /50x.html;

    location = /50x.html {
        root /usr/share/nginx/html;
    }

    # pass the PHP scripts to FastCGI server listening on php-fpm:9000
    #
    location ~ \.php$ {
		fastcgi_split_path_info  ^(.+\.php)(.*)$;

        # let bootstrap file catch the calls to unexising PHP files
        set $fsn /$bootstrap;

        if (-f $document_root$fastcgi_script_name){
            set $fsn $fastcgi_script_name;
        }

        fastcgi_pass   php-fpm:9000;
        include        fastcgi_params;
        fastcgi_param  SCRIPT_FILENAME  $document_root$fsn;

        # PATH_INFO and PATH_TRANSLATED can be omitted, but RFC 3875 specifies them for CGI
        fastcgi_param  PATH_INFO        $fastcgi_path_info;
        fastcgi_param  PATH_TRANSLATED  $document_root$fsn;
    }

    # prevent nginx from serving dotfiles (.htaccess, .svn, .git, etc.)
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }
}
